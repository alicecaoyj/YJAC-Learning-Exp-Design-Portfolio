    // Set your passcode here

    function checkArticulatePasscode() {

      const input = document.getElementById('passcode-input').value;

      const message = document.getElementById('message');

      if (input === articulatePasscode) {

        // Show the Articulate content section

        document.getElementById('articulate-access').style.display = 'block';

        // Hide the passcode form

        document.getElementById('passcode-form').style.display = 'none';

        message.textContent = ""; // Clear any previous messages

        // Autoplay the video

        document.getElementById('articulate-video').src += "?autoplay=1";

      } else {

        message.textContent = "Incorrect passcode. Please try again.";

      }

    }

    // Prevent right-click

    document.addEventListener('contextmenu', function(event) {

      event.preventDefault();

    });

    // Modal logic for resume request

    var modal = document.getElementById("resumeModal");

    var span = document.getElementsByClassName("close")[0];

    span.onclick = function() {

      modal.style.display = "none";

    }

    window.onclick = function(event) {

      if (event.target == modal) {

        modal.style.display = "none";

      }

    }

    window.onload = function() {

      modal.style.display = "block";

    }
